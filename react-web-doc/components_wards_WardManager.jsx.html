/**
 * WardManager component for managing wards within a specific hospital block.
 * Allows viewing, adding, deleting, and searching wards.
 * Wards are grouped and displayed by floor, with collapsible sections.
 */

import React, { useEffect, useState } from 'react';
// useParams hook to access route parameters (blockId)
// useNavigate hook for programmatic navigation
// useAuth hook to access authentication context (apiService, hospital)
import { useParams, useNavigate } from 'react-router-dom';
import { useAuth } from '../../hooks/useAuth';
import LoadingSpinner from '../common/LoadingSpinner';
import toast from 'react-hot-toast';

const WardManager = () => {
  const { blockId } = useParams();
  const { apiService, hospital } = useAuth();
  const navigate = useNavigate();

  // State for storing the current block details
  const [block, setBlock] = useState(null);
  // State for storing the list of wards in the block
  const [wards, setWards] = useState([]);
  // State for the new ward floor input
  const [floor, setFloor] = useState('');
  // State for the new ward name input
  const [name, setName] = useState('');
  // State for the search input
  const [search, setSearch] = useState('');
  // State for loading indicator
  const [loading, setLoading] = useState(true);
  // State for error messages
  const [error, setError] = useState('');
  // State to manage expanded/collapsed state of floor sections
  const [expanded, setExpanded] = useState({});

  /**
 * Effect hook to fetch block and ward data when the component mounts
 * or blockId, hospital.id, or apiService changes.
 */
  useEffect(() => {
    const fetchData = async () => {
      try {
        setLoading(true);
        // Fetch block details
        const b = await apiService.getBlock(hospital.id, blockId);
        // Fetch wards belonging to the block
        const w = await apiService.getWardsByBlock(hospital.id, blockId);
        setBlock(b);
        // Sort wards by floor number initially
        setWards(w);
        setLoading(false);
      } catch (err) {
        setError('Failed to load wards or block');
        setLoading(false);
      }
    };

    fetchData();
  }, [blockId, hospital.id, apiService]);

  /**
 * Handles adding a new ward.
 * Validates the floor number against the block's total floors.
 * Calls the API to create the ward and updates the state.
 */
  const handleAddWard = async () => {
    const floorNum = parseInt(floor);
    // Validate floor number
    if (floorNum < 1 || floorNum > block.no_of_floors) {
      return;
    }
    try {
      const newWard = await apiService.createWard(hospital.id, { ward_name:name, floor: floorNum, block: blockId });
      setWards(prev => [...prev, newWard]);
      setName('');
      setFloor('');
      setError('');
      toast.success('Ward added successfully!');
    } catch (err) {
      toast.error(err.message || 'Failed to add ward');
    }
  };

  /**
 * Handles deleting a ward by its ID.
 * Calls the API to delete the ward and updates the state.
 */
  const handleDelete = async (id) => {
    try {
      await apiService.deleteWard(hospital.id, id);
      setWards(prev => prev.filter(w => w.id !== id));
      toast.success('Ward deleted successfully!');
    } catch(err) {
      toast.error(err.message || 'Failed to delete ward');
    }
  };

  const toggleFloor = (floorNum) => {
    setExpanded(prev => ({ ...prev, [floorNum]: !prev[floorNum] }));
  };

  /**
 * Groups wards by their floor number.
 * Filters wards based on the search query before grouping.
 * @returns {object} An object where keys are floor numbers and values are arrays of ward objects.
 */
  const groupedWards = () => {
    const filtered = search
      ? wards.filter(w => w.ward_name.toLowerCase().includes(search.toLowerCase()))
      : wards;

    return filtered.reduce((acc, ward) => {
      acc[ward.floor] = acc[ward.floor] || [];
      acc[ward.floor].push(ward);
      return acc;
    }, {});
  };

  // Display loading spinner while fetching data
  if (loading) return <LoadingSpinner />;

  return (
    <div className="max-w-4xl mx-auto p-4 space-y-6">
      <div className="flex justify-between items-center">
        <h2 className="text-2xl font-bold">Wards in {block?.name}</h2>
        <button onClick={() => navigate(-1)} className="text-gray-500 hover:underline">
          ← Back
        </button>
      </div>

      {/* Add Ward Inputs */}
      <div className="flex flex-col md:flex-row gap-2">
        <input
          type="text"
          placeholder="Ward Name"
          value={name}
          onChange={(e) => setName(e.target.value)}
          className="border border-gray-300 px-3 py-2 rounded w-full md:w-1/3"
        />
        <input
          type="number"
          placeholder="Floor"
          value={floor}
          onChange={(e) => setFloor(e.target.value)}
          className="border border-gray-300 px-3 py-2 rounded w-full md:w-1/6"
        />
        {/* Add Ward Button */}
        <button
          onClick={handleAddWard}
          className="bg-blue-600 text-white px-4 py-2 rounded hover:bg-blue-700 w-full md:w-auto"
        >
          Add Ward
        </button>
      </div>

      {/* Search Box */}
      <input
        type="text"
        placeholder="Search ward by name..."
        value={search}
        onChange={(e) => setSearch(e.target.value)}
        className="w-full px-3 py-2 border border-gray-300 rounded"
      />

      {error && <p className="text-red-600">{error}</p>}


      {/* Wards Grouped by Floor */}
      {Object.entries(groupedWards())
        .sort(([a], [b]) => parseInt(a) - parseInt(b))
        .map(([floorNum, wardList]) => (
          <div key={floorNum} className="border rounded shadow-sm bg-white">
            <button
              onClick={() => toggleFloor(floorNum)}
              className="w-full flex justify-between items-center px-4 py-3 bg-gray-100 text-left font-medium hover:bg-gray-200"
            >
              {/* Floor header with ward count and expand/collapse icon */}
              <span>Floor {floorNum} ({wardList.length} ward{wardList.length !== 1 ? 's' : ''})</span>
              <span>{expanded[floorNum] ? '▴' : '▾'}</span>
            </button>
            {expanded[floorNum] && (
              <ul className="px-4 py-2 space-y-2">
                {wardList.map(ward => (
                  <li key={ward.id} className="flex justify-between items-center border-b py-2">
                    {/* Ward name and Delete button */}
                    <span>{ward.ward_name}</span>
                    <button
                      onClick={() => handleDelete(ward.id)}
                      className="text-red-600 text-sm hover:underline"
                    >
                      Delete
                    </button>
                  </li>
                ))}
              </ul>
            )}
          </div>
        ))}

      {/* Message when no wards are found (after search or initially empty) */}
      {Object.keys(groupedWards()).length === 0 && (
        <div className="text-gray-500 text-center">No wards found.</div>
      )}
    </div>
  );
};

export default WardManager;
